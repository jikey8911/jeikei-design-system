NeuralGrid.jsx
import { useRef, useEffect } from "react"
import { startNeuralAnimation } from "./neuralCanvas"

export default function NeuralGrid({ sparkles=true }) {

  const canvasRef = useRef()

  useEffect(()=>{

    if(!sparkles) return

    const canvas = canvasRef.current
    const cleanup = startNeuralAnimation(canvas)

    return cleanup

  },[sparkles])

  return(

  <div className="neural-bg">

      <canvas
      ref={canvasRef}
      className="neural-canvas"
      />

  </div>

  )
}
2️⃣ Motor de destellos neuronales
neuralCanvas.js

Esto crea pulsos que viajan por la malla.

export function startNeuralAnimation(canvas){

const ctx = canvas.getContext("2d")

let width
let height

function resize(){

width = canvas.width = window.innerWidth
height = canvas.height = window.innerHeight

}

resize()

window.addEventListener("resize",resize)


const nodes = []
const sparks = []

const grid = 80


for(let x=0;x<width;x+=grid){

for(let y=0;y<height;y+=grid){

nodes.push({x,y})

}

}


function createSpark(){

const start = nodes[Math.floor(Math.random()*nodes.length)]
const end = nodes[Math.floor(Math.random()*nodes.length)]

sparks.push({
x:start.x,
y:start.y,
tx:end.x,
ty:end.y,
progress:0
})

}


setInterval(createSpark,700)


function animate(){

ctx.clearRect(0,0,width,height)

ctx.strokeStyle="rgba(0,255,200,0.08)"

nodes.forEach(n=>{

ctx.beginPath()
ctx.arc(n.x,n.y,1,0,Math.PI*2)
ctx.fillStyle="rgba(0,255,200,0.2)"
ctx.fill()

})

sparks.forEach(s=>{

s.progress+=0.02

s.x += (s.tx-s.x)*0.05
s.y += (s.ty-s.y)*0.05

ctx.beginPath()

src/neural/NeuralEngine.ts
type Node = {
 id:number
 x:number
 y:number
}

type Pulse = {
 from:number
 to:number
 progress:number
}

export class NeuralEngine {

nodes:Node[]=[]
pulses:Pulse[]=[]

constructor(count=40){

for(let i=0;i<count;i++){

this.nodes.push({
id:i,
x:Math.random()*100,
y:Math.random()*100
})

}

}

createPulse(){

const a=Math.floor(Math.random()*this.nodes.length)
const b=Math.floor(Math.random()*this.nodes.length)

this.pulses.push({
from:a,
to:b,
progress:0
})

}

tick(){

this.pulses=this.pulses.map(p=>({
...p,
progress:p.progress+0.02
})).filter(p=>p.progress<1)

}

}

const canvasRef=useRef<HTMLCanvasElement>(null)
const engine=new NeuralEngine()

useEffect(()=>{

const canvas=canvasRef.current!
const ctx=canvas.getContext("2d")!

function frame(){

ctx.clearRect(0,0,canvas.width,canvas.height)

engine.tick()

engine.nodes.forEach(n=>{
ctx.fillRect(n.x*10,n.y*10,2,2)
})

requestAnimationFrame(frame)

}

frame()

},[])

return <canvas ref={canvasRef} className="absolute inset-0"/>
}
